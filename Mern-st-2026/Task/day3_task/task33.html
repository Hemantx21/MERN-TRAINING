let grade = prompt("Enter your grade (A, B, C, D, or F):").toUpperCase(); switch (grade) { case "A": console.log("Excellent"); break; case "B": console.log("Very Good"); break; case "C": console.log("Good"); break; case "D": console.log("Needs Improvement"); break; case "F": console.log("Fail"); break; default: console.log("Invalid Grade"); }
